\frame{\frametitle{Stability criteria of the simple model can be extended for complex cycles}
    \begin{adjustbox}{max totalsize={\textwidth}{0.6\textheight},center}
      \input{complex-autocatalytic-cycles.tex}
    \end{adjustbox}
    \pause
    Sufficient condition for stability is:
    \begin{equation*}
        \forall_i \quad \beta_i \geq \alpha_i
    \end{equation*}
    Where $\alpha_i=\frac{df_{a_i}}{dX_i}$ and $\beta_i=\frac{df_{b_i}}{dX_i}$ 
}

\frame{\frametitle{Theoretical $\beta_i\geq\alpha_i$ constraint can be converted to experimental prediction on reaction saturation level}
    %\begin{adjustbox}{max totalsize={\textwidth}{0.8\textheight},center}
      %\input{derivative-saturation-connection.tex}
    %\end{adjustbox}
    If reactions are 'nice' concave functions, sufficient condition for a steady state to be stable is that at all branch points the branch reaction is less saturated than the autocatalytic reaction.
}

\frame{\frametitle{Conclusions}
\begin{itemize}
    \item Autocatalytic cycles play a major role in central carbon metabolism
    \item Proper function of autocatalytic cycles depends on kinetic parameters of enzymes
    \begin{itemize}
        \item May require sub-optimal kinetic parameters
    \end{itemize}
    \item In metabolic engineering of autocatalytic cycles, native kinetic parameters might prohibit function
    \item Stability of autocatalytic cycles depends on under-saturation of branch reactions
    \item Fluxomics data reveals sub-optimality constraints are maintained in-vivo
\end{itemize}
}

\frame{\frametitle{Outlook}
\begin{itemize}
    \item Efficient algorithm for identification of autocatalytic cycles in large metabolic networks
    \item Tighter limits on kinetic parameters
    \item Experimental exploration of different autocatalytic cycles function in-vivo
\end{itemize}
